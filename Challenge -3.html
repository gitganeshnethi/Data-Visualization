from ipywidgets import widgets,interact
funds = widgets.Dropdown(options = ['All']+list(fund['etf_name'].unique()),
                        value = 'All',
                         description = "Select the fund")
company_name=widgets.Dropdown(options = ['All']+list(fund['constituent_name'].unique()),
                        value = 'All',
                        description = "Select the company")
def plotit(x,y):
    if (funds.value!='All') & (company_name.value!='All') :
        df2 = fund[(fund.etf_name ==funds.value) & (fund.constituent_name == company_name.value)]
    elif(funds.value=='All') and (company_name.value!='All') :
        df2 = fund[(fund.constituent_name ==company_name.value)]
    elif(funds.value!='All') and (company_name.value=='All') :
        df2 = fund[(fund.etf_name ==funds.value)]
    else:
        df2 = fund
    #print(df2)
    cnt= df2.groupby(df2.month)[['weight_percentage']].mean().reset_index()
    
    
    plt.rcParams['figure.figsize']=[15,5]
    plt.bar(cnt['month'],cnt['weight_percentage'],color='red')
    #x.set_xticks(['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Oct','Nov','Dec'])
interact(plotit,x=funds,y =company_name)
